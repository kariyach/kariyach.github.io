# LDAモデルを作成して tidy で beta を取得
top_terms <- dtm_1740_1742 |>
  LDA(k = 5, control = list(seed = 1234)) |>
  tidy(matrix = "beta") |>                 # トピックごとの単語確率
  group_by(topic) |>
  slice_max(beta, n = 5) |>
  ungroup() |>
  arrange(topic, -beta) |>
  mutate(term = fct_reorder(term, beta))   # 棒グラフ用に順序調整

# ggplotで可視化
top_terms |>
  ggplot(aes(x = beta, y = term, fill = factor(topic))) +
  geom_col(show.legend = FALSE) +
  facet_wrap(~ topic, scales = "free_y") +
  labs(x = "Beta (Probability)", y = "Term", title = "Top 5 Terms per Topic")
