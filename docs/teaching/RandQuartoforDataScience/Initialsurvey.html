# アンケート結果のデータフレームの例
survey_data <- tibble(
  ID = 1:5,
  Q1 = c("A", "B", "A", "C", "B"),
  Q2 = c("Yes", "No", "Yes", "Yes", "No"),
  Q3 = c(5, 4, 3, 5, 2),
  Q4 = c("Male", "Female", "Female", "Male", "Male"),
  Q5 = c("Employed", "Unemployed", "Employed", "Employed", "Unemployed")
)

# 各質問のデータをtibbleとしてリストに格納
data_list <- list(
  Q1 = survey_data %>% select(ID, Q1),
  Q2 = survey_data %>% select(ID, Q2),
  Q3 = survey_data %>% select(ID, Q3),
  Q4 = survey_data %>% select(ID, Q4),
  Q5 = survey_data %>% select(ID, Q5)
)

# 関数の引数名を df に変更
create_bar_plot <- function(df, question) {
  ggplot(df, aes_string(x = question)) +
    geom_bar() +
    theme_minimal() +
    labs(title = paste("Distribution of", question))
}

# 各質問に対して棒グラフを作成
plots <- map2(data_list, names(data_list), create_bar_plot)

# すべてのプロットを表示
for (plot in plots) {
  print(plot)
}

df_初回アンケート %>%
  group_by(学域) %>% 
  summarise(
    人数 = n()
  ) %>%
  mutate(
    比率 = 人数/sum(人数)
  ) %>%
  ungroup() %>%
  ggplot(aes(x = "", y = 人数, group = 学域)) +
  geom_col(
    aes(color = 学域, fill = after_scale(alpha(color, 0.9))),
    position = position_fill(),
  ) +
  geom_text(
    aes(label = str_c(scales::percent(比率, accuracy = 0.1), "(", 人数, "人)")),
    position = position_fill(reverse = FALSE, vjust = 0.5),
  ) +
  scale_y_continuous(
    expand = expansion(mult = c(0.01, 0.01)),
    labels = scales::label_percent()) +
  labs(
    x = "",
    y = "",
    color = "学域"
  ) +
  coord_flip() +
  scale_color_paletteer_d("DresdenColor::briefcases") #配色変更
